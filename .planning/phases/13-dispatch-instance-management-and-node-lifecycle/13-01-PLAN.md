**security/ratelimit.go changes:**
Add `ProjectRateLimiter` struct BELOW the existing `ChannelRateLimiter` (keep old struct for backward compat):

```go
type ProjectRateLimiter struct {
    mu       sync.Mutex
    limiters map[string]*rate.Limiter
    limit    rate.Limit
    burst    int
}

func NewProjectRateLimiter(requestsPerWindow, windowSeconds int) *ProjectRateLimiter {
    r := rate.Limit(float64(requestsPerWindow) / float64(windowSeconds))
    return &ProjectRateLimiter{
        limiters: make(map[string]*rate.Limiter),
        limit:    r,
        burst:    requestsPerWindow,
    }
}

func (p *ProjectRateLimiter) Allow(project string) bool {
    p.mu.Lock()
    l, ok := p.limiters[project]
    if !ok {
        l = rate.NewLimiter(p.limit, p.burst)
        p.limiters[project] = l
    }
    p.mu.Unlock()
    return l.Allow()
}
```

Note: `ProjectRateLimiter.Allow()` returns just `bool` (not `(bool, time.Duration)`) because the dispatcher only needs allow/deny — no delay reporting needed.
